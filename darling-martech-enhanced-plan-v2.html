import * as cheerio from 'cheerio'

export interface AuditResult {
  url: string
  score: number
  checks: CheckResult[]
  summary: string
}

export interface CheckResult {
  id: string
  label: string
  status: 'pass' | 'warn' | 'fail'
  weight: number
  message: string
  fix?: string
}

export async function runAudit(url: string): Promise<AuditResult> {
  const checks: CheckResult[] = []

  // 1. Fetch the page
  const html = await fetch(url).then(r => r.text())
  const $ = cheerio.load(html)

  // 2. Check robots.txt for AI bot permissions
  const robotsResult = await checkRobots(url)
  checks.push(robotsResult)

  // 3. Check for Schema.org structured data
  const schemas = $('script[type="application/ld+json"]')
  checks.push({
    id: 'schema-markup',
    label: 'Schema Markup',
    status: schemas.length > 0 ? 'pass' : 'fail',
    weight: 20,
    message: schemas.length > 0
      ? `Found ${schemas.length} schema blocks`
      : 'No JSON-LD schema found',
    fix: schemas.length === 0
      ? 'Add at minimum: LocalBusiness, Service, and FAQ schema'
      : undefined,
  })

  // 4. Check heading hierarchy (H1 → H2 → H3)
  const h1Count = $('h1').length
  checks.push({
    id: 'heading-hierarchy',
    label: 'Heading Hierarchy',
    status: h1Count === 1 ? 'pass' : h1Count === 0 ? 'fail' : 'warn',
    weight: 15,
    message: h1Count === 1
      ? 'Single H1 found — good structure'
      : `Found ${h1Count} H1 tags — AI models prefer single H1`,
    fix: h1Count !== 1 ? 'Use exactly one H1 per page as the primary topic signal' : undefined,
  })

  // 5. Check for FAQ content patterns
  const faqPatterns = $('[class*="faq"], [id*="faq"], details, summary').length
  checks.push({
    id: 'faq-content',
    label: 'FAQ / Q&A Content',
    status: faqPatterns > 0 ? 'pass' : 'warn',
    weight: 15,
    message: faqPatterns > 0
      ? 'FAQ content detected — AI models love Q&A format'
      : 'No FAQ content detected',
    fix: 'Add 3-5 Q&A pairs per service page + FAQ schema markup',
  })

  // Calculate weighted score
  const score = calculateScore(checks)

  return { url, score, checks, summary: buildSummary(score) }
}

function calculateScore(checks: CheckResult[]): number {
  const totalWeight = checks.reduce((sum, c) => sum + c.weight, 0)
  const earned = checks.reduce((sum, c) => {
    if (c.status === 'pass') return sum + c.weight
    if (c.status === 'warn') return sum + (c.weight * 0.5)
    return sum
  }, 0)
  return Math.round((earned / totalWeight) * 100)
}

export interface Touchpoint {
  channel: string       // 'google-ads' | 'email' | 'organic' | 'direct' | 'social'
  timestamp: Date
  sessionId: string
  converted: boolean
  revenue?: number
}

export interface AttributionResult {
  model: string
  channelCredits: Record<string, number>  // channel → revenue credit %
  insights: string[]
}

// Last-touch: 100% credit to final touchpoint before conversion
export function lastTouch(touchpoints: Touchpoint[]): AttributionResult {
  const credits: Record<string, number> = {}
  const conversions = touchpoints.filter(t => t.converted)
  conversions.forEach(t => {
    credits[t.channel] = (credits[t.channel] || 0) + (t.revenue || 1)
  })
  return { model: 'Last Touch', channelCredits: normalize(credits), insights: [] }
}

// Linear: equal credit split across all touchpoints in journey
export function linear(journeys: Touchpoint[][]): AttributionResult {
  const credits: Record<string, number> = {}
  journeys.filter(j => j.some(t => t.converted)).forEach(journey => {
    const revenue = journey.find(t => t.converted)?.revenue || 1
    const share = revenue / journey.length
    journey.forEach(t => {
      credits[t.channel] = (credits[t.channel] || 0) + share
    })
  })
  return { model: 'Linear', channelCredits: normalize(credits), insights: [] }
}

// Time-decay: recent touchpoints get more credit (half-life: 7 days)
export function timeDecay(journeys: Touchpoint[][]): AttributionResult {
  const credits: Record<string, number> = {}
  journeys.filter(j => j.some(t => t.converted)).forEach(journey => {
    const conversion = journey.find(t => t.converted)!
    const revenue = conversion.revenue || 1
    let totalDecay = 0
    const decays = journey.map(t => {
      const daysDiff = (conversion.timestamp.getTime() - t.timestamp.getTime()) / (86400000)
      return Math.pow(0.5, daysDiff / 7)  // 7-day half life
    })
    totalDecay = decays.reduce((a, b) => a + b, 0)
    journey.forEach((t, i) => {
      const share = (decays[i] / totalDecay) * revenue
      credits[t.channel] = (credits[t.channel] || 0) + share
    })
  })
  return { model: 'Time Decay', channelCredits: normalize(credits), insights: [] }
}

function normalize(credits: Record<string, number>): Record<string, number> {
  const total = Object.values(credits).reduce((a, b) => a + b, 0)
  return Object.fromEntries(
    Object.entries(credits).map(([k, v]) => [k, Math.round((v / total) * 100) / 100])
  )
}

// Overlap detection logic
export function detectOverlaps(selectedIds: string[]): OverlapWarning[] {
  const selected = MARTECH_TOOLS.filter(t => selectedIds.includes(t.id))
  const warnings: OverlapWarning[] = []
  selected.forEach(tool => {
    const overlapping = tool.alternatives?.filter(alt => selectedIds.includes(alt)) || []
    overlapping.forEach(altId => {
      const alt = MARTECH_TOOLS.find(t => t.id === altId)
      if (alt) warnings.push({
        tools: [tool.name, alt.name],
        message: `${tool.name} and ${alt.name} both handle ${tool.subcategory}`,
        recommendation: `Pick one. You're likely paying for both unnecessarily.`,
        monthlySavings: Math.min(tool.avgMonthlyCost || 0, alt.avgMonthlyCost || 0),
      })
    })
  })
  return warnings
}

export function scoreLeads(
  leads: Record<string, string | number>[],
  model: ScoringModel
): ScoredLead[] {
  return leads.map(lead => {
    let score = 0
    const matchedRules: string[] = []
    model.rules.forEach(rule => {
      const val = lead[rule.field]
      if (typeof val === 'undefined') return
      let matched = false
      if (rule.operator === 'equals') matched = String(val) === String(rule.value)
      if (rule.operator === 'contains') matched = String(val).toLowerCase().includes(String(rule.value).toLowerCase())
      if (rule.operator === 'greater_than') matched = Number(val) > Number(rule.value)
      if (rule.operator === 'less_than') matched = Number(val) < Number(rule.value)
      if (matched) { score += rule.points; matchedRules.push(rule.label) }
    })
    const tier = model.tiers.find(t => score >= t.min) || model.tiers[model.tiers.length - 1]
    return { ...lead, _score: Math.min(score, model.maxScore), _tier: tier.label, _matchedRules: matchedRules }
  }).sort((a, b) => b._score - a._score)
}

export interface SecurityCheck {
  id: string
  label: string
  severity: 'critical' | 'warning' | 'pass'
  message: string
  fix?: string
}

export async function runFortressScan(domain: string): Promise<SecurityCheck[]> {
  const url = `https://${domain}`
  const checks: SecurityCheck[] = []

  try {
    const res = await fetch(url, { method: 'HEAD', redirect: 'manual' })
    const headers = res.headers

    // HTTPS enforcement
    checks.push({
      id: 'https',
      label: 'HTTPS Enforcement',
      severity: url.startsWith('https') ? 'pass' : 'critical',
      message: url.startsWith('https') ? 'HTTPS active' : 'Site not served over HTTPS',
      fix: 'Enable HTTPS via Cloudflare SSL settings',
    })

    // HSTS header
    const hsts = headers.get('strict-transport-security')
    checks.push({
      id: 'hsts',
      label: 'HTTP Strict Transport Security',
      severity: hsts ? 'pass' : 'warning',
      message: hsts ? `HSTS enabled: ${hsts}` : 'HSTS header missing',
      fix: 'Add Strict-Transport-Security header with max-age=31536000',
    })

    // X-Frame-Options
    const xfo = headers.get('x-frame-options')
    checks.push({
      id: 'x-frame-options',
      label: 'Clickjacking Protection',
      severity: xfo ? 'pass' : 'warning',
      message: xfo ? `X-Frame-Options: ${xfo}` : 'Clickjacking protection missing',
      fix: 'Add X-Frame-Options: SAMEORIGIN header',
    })

    // Server version disclosure
    const server = headers.get('server')
    const discloses = server && !server.toLowerCase().includes('cloudflare')
    checks.push({
      id: 'server-disclosure',
      label: 'Server Version Disclosure',
      severity: discloses ? 'warning' : 'pass',
      message: discloses ? `Server header exposes: ${server}` : 'Server version not exposed',
      fix: 'Route traffic through Cloudflare to mask server identity',
    })

    // CSP header
    const csp = headers.get('content-security-policy')
    checks.push({
      id: 'csp',
      label: 'Content Security Policy',
      severity: csp ? 'pass' : 'critical',
      message: csp ? 'CSP header present' : 'No Content Security Policy found',
      fix: 'Implement CSP to prevent XSS attacks — start with report-only mode',
    })

  } catch (e) {
    checks.push({ id: 'connection', label: 'Site Reachability', severity: 'critical', message: 'Could not connect to site' })
  }

  return checks
}
